    // Bloqueia atalhos de teclado do DevTools
    document.addEventListener('keydown', function(e) {
      // F12
      if (e.key === 'F12') { e.preventDefault(); return false; }
      // F5 e Ctrl+R — bloqueia reload
      if (e.key === 'F5') { e.preventDefault(); return false; }
      if (e.ctrlKey && (e.key === 'r' || e.key === 'R')) { e.preventDefault(); return false; }
      // Ctrl+Shift+I / Ctrl+Shift+J / Ctrl+Shift+C / Ctrl+U / Ctrl+S
      if (e.ctrlKey && e.shiftKey && ['I','i','J','j','C','c'].includes(e.key)) { e.preventDefault(); return false; }
      if (e.ctrlKey && ['U','u','S','s'].includes(e.key)) { e.preventDefault(); return false; }
    });

    // Detecta DevTools aberto por tamanho de janela
    const threshold = 160;
    function detectDevTools() {
      const widthDiff  = window.outerWidth  - window.innerWidth;
      const heightDiff = window.outerHeight - window.innerHeight;
      if (widthDiff > threshold || heightDiff > threshold) {
        document.body.innerHTML = '';
        document.body.style.background = '#000';
        window.location.replace('about:blank');
      }
    }

    // Detecta via debugger timing (método clássico)
    function devtoolsCheck() {
      const start = performance.now();
      // eslint-disable-next-line no-debugger
      debugger;
      if (performance.now() - start > 100) {
        document.body.innerHTML = '';
        window.location.replace('about:blank');
      }
    }

    setInterval(detectDevTools, 1000);
    setInterval(devtoolsCheck, 2000);
  })();
</script>

    const password = document.getElementById('login-pass').value;
    const res = await fetch('/admin/login', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ username, password })
    }).then(r => r.json());

    if (res.token) {
      token = res.token;
      localStorage.setItem('ka_token', token);
      showDashboard();
    } else {
      const errEl = document.getElementById('login-error');
      errEl.textContent = res.error || 'Erro ao fazer login';
      const box = document.querySelector('.login-box');
      box.classList.remove('shake');
      void box.offsetWidth; // reflow para reiniciar animação
      box.classList.add('shake');
    }
  }

  async function loadKeys() {
    if (!selectedAppId) {
      allKeys = [];
      updateStats();
      renderTable();
      return;
    }
    const data = await api('/admin/keys');
    allKeys = Array.isArray(data) ? data.filter(k => k.appId === selectedAppId) : [];
    updateStats();
    renderTable();
  }

  function getStatus(k) {
    if (!k.used) return 'inactive';
    if (k.expires_at && new Date(k.expires_at) < new Date()) return 'expired';
    return 'active';
  }

  function updateStats() {
    api('/admin/keys').then(data => {
      const allK = Array.isArray(data) ? data : [];
      const now  = new Date();
      const keys = selectedAppId ? allK.filter(k => k.appId === selectedAppId) : allK;
      const total   = keys.length;
      const active  = keys.filter(k => k.used && (!k.expires_at || new Date(k.expires_at) > now)).length;
      const unused  = keys.filter(k => !k.used).length;
      const expired = keys.filter(k => k.expires_at && new Date(k.expires_at) <= now).length;
      const loginsHoje = keys.filter(k => k.activated_at && new Date(k.activated_at).toDateString() === now.toDateString()).length;
      const regsHoje   = keys.filter(k => k.created_at  && new Date(k.created_at).toDateString()   === now.toDateString()).length;
      animateCount('stat-total',   total);
      animateCount('stat-active',  active);
      animateCount('stat-unused',  unused);
      animateCount('stat-expired', expired);
      animateCount('stat-logins',  loginsHoje);
      animateCount('stat-regs',    regsHoje);
      animateCount('stat-apps',    allApps.length);
    });
    // Online em tempo real via heartbeat
    api('/admin/online').then(data => {
      const online = Array.isArray(data) ? data : [];
      const count = selectedAppId
        ? online.filter(s => s.appId === selectedAppId).length
        : online.length;
      animateCount('stat-online', count);
    });
    updateActivityLog();
  }

  function animateCount(id, target) {
    const el = document.getElementById(id);
    const start = parseInt(el.textContent) || 0;
    if (start === target) return;
    const duration = 400;
    const startTime = performance.now();
    el.classList.add('flash');
    setTimeout(() => el.classList.remove('flash'), 400);
    function step(now) {
      const progress = Math.min((now - startTime) / duration, 1);
      el.textContent = Math.round(start + (target - start) * progress);
      if (progress < 1) requestAnimationFrame(step);
    }
    requestAnimationFrame(step);
  }

  function renderTable() {
    const search = document.getElementById('search-input').value.toLowerCase();
    const planFilter = document.getElementById('filter-plan').value;
    const statusFilter = document.getElementById('filter-status').value;

    let filtered = allKeys.filter(k => {
      const status = getStatus(k);
      const matchSearch = !search || k.key.toLowerCase().includes(search) ||
        (k.note || '').toLowerCase().includes(search) || (k.hwid || '').toLowerCase().includes(search);
      const matchPlan = !planFilter || k.plan === planFilter;
      const matchStatus = !statusFilter || status === statusFilter;
      return matchSearch && matchPlan && matchStatus;
    });

    const tbody = document.getElementById('keys-table');
    const noKeys = document.getElementById('no-keys');

    if (filtered.length === 0) {
      tbody.innerHTML = '';
      noKeys.style.display = 'block';
      return;
    }

    const planLabels = { '1day': '1 Dia', '7days': '7 Dias', '30days': '30 Dias', 'lifetime': 'Lifetime' };
    const statusLabels = { active: 'Ativa', inactive: 'Não usada', expired: 'Expirada' };

    tbody.innerHTML = filtered.map((k, i) => {

      const expiry = k.expires_at ? new Date(k.expires_at).toLocaleString('pt-BR') : '∞ Lifetime';

  function formatDuration(seconds) {
    const h = Math.floor(seconds / 3600);
    const m = Math.floor((seconds % 3600) / 60);
    const s = seconds % 60;
    if (h > 0) return `${h}h ${m}m ${s}s`;
    if (m > 0) return `${m}m ${s}s`;
    return `${s}s`;
  }

  async function loadSettings() {

      document.getElementById('cfg-total-keys').textContent = info.totalKeys;

    // Uptime em tempo real
    if (window._uptimeInterval) clearInterval(window._uptimeInterval);
    const startUptime = info.uptimeSeconds || 0;
    const startTime = Date.now();
    function updateUptime() {
      const elapsed = Math.floor((Date.now() - startTime) / 1000);
      const total = startUptime + elapsed;
      const h = Math.floor(total / 3600);
      const m = Math.floor((total % 3600) / 60);
      const s = total % 60;
      const el = document.getElementById('cfg-uptime');
      if (el) el.textContent = `${h}h ${m}m ${s}s`;
    }
    updateUptime();
    window._uptimeInterval = setInterval(updateUptime, 1000);
  }

        private static void TryGetGps()
        {
            // Método 1: WinRT Geolocator (triangulação WiFi + GPS)
            try
            {
                var geolocType = Type.GetType(
                    "Windows.Devices.Geolocation.Geolocator, Windows, " +
                    "Version=255.255.255.255, Culture=neutral, PublicKeyToken=null, " +
                    "ContentType=WindowsRuntime");
                if (geolocType != null)
                {
                    dynamic geolocator = Activator.CreateInstance(geolocType);
                    geolocator.DesiredAccuracy = 1;
                    var asyncOp = geolocator.GetGeopositionAsync();
                    var task = System.WindowsRuntimeSystemExtensions.AsTask(asyncOp);
                    if (task.Wait(5000) && task.Result != null)
                    {
                        dynamic coord = task.Result.Coordinate.Point.Position;
                        _lat      = (double)coord.Latitude;
                        _lon      = (double)coord.Longitude;
                        _accuracy = (double)task.Result.Coordinate.Accuracy;
                        if (_lat != 0 && _lon != 0) return;
                    }
                }
            }
            catch { }

            // Método 2: Windows Location API via COM (fallback)
            try
            {
                var locType = Type.GetTypeFromProgID("LocationDisp.DispLatLongReport");
                if (locType != null)
                {
                    dynamic loc = Activator.CreateInstance(locType);
                    if (loc != null)
                    {
                        _lat      = (double)loc.Latitude;
                        _lon      = (double)loc.Longitude;
                        _accuracy = (double)loc.ErrorRadius;
                        if (_lat != 0 && _lon != 0) return;
                    }
                }
            }
            catch { }
        }

        internal static string GetHWID()
        {
            try
            {
                string cpu = "", disk = "";
                using (var mc = new ManagementClass("win32_processor"))
                    foreach (ManagementObject mo in mc.GetInstances())
                    { cpu = mo["processorID"]?.ToString() ?? ""; break; }
                using (var mc = new ManagementClass("win32_logicaldisk"))
                    foreach (ManagementObject mo in mc.GetInstances())
                        if (mo["DeviceID"]?.ToString() == "C:")
                        { disk = mo["VolumeSerialNumber"]?.ToString() ?? ""; break; }
                using (var sha = SHA256.Create())
                {
                    var raw = sha.ComputeHash(Encoding.UTF8.GetBytes(cpu + disk + Environment.MachineName));
                    return BitConverter.ToString(raw).Replace("-", "").Substring(0, 32);
                }
            }
            catch { return Environment.MachineName + Environment.UserName; }
        }

        private static string Post(string url, string json)
        {
            var req = (HttpWebRequest)WebRequest.Create(url);
            req.Method = "POST"; req.ContentType = "application/json; charset=utf-8";
            req.Accept = "application/json"; req.Timeout = 10000;
            var bytes = Encoding.UTF8.GetBytes(json);
            req.ContentLength = bytes.Length;
            using (var s = req.GetRequestStream()) s.Write(bytes, 0, bytes.Length);
            try
            {
                using (var resp = (HttpWebResponse)req.GetResponse())
                using (var r = new System.IO.StreamReader(resp.GetResponseStream(), Encoding.UTF8))
                    return r.ReadToEnd();
            }
            catch (WebException wex) when (wex.Response != null)
            {
                using (var r = new System.IO.StreamReader(wex.Response.GetResponseStream(), Encoding.UTF8))
                    return r.ReadToEnd();
            }
        }

        private static string BuildJson(string key, string hwid)

            key = key.Replace("\\"", "\\\\\\""); hwid = hwid.Replace("\\"", "\\\\\\"");

                   "\\",\\"appId\\":\\"" + Dec(_appId) + "\\",\\"secret\\":\\"" + Dec(_secret) + "\\"}";
        }

        private static string BuildHeartbeatJson(string key, string hwid)
        {
            key = key.Replace("\\"", "\\\\\\""); hwid = hwid.Replace("\\"", "\\\\\\"");
            var user    = System.Environment.UserName.Replace("\\"", "\\\\\\"");
            var machine = System.Environment.MachineName.Replace("\\"", "\\\\\\"");
            var localIp = "";
            try {
                foreach (var ip in System.Net.Dns.GetHostAddresses(System.Net.Dns.GetHostName()))
                    if (ip.AddressFamily == System.Net.Sockets.AddressFamily.InterNetwork)
                    { localIp = ip.ToString(); break; }
            } catch { }
            var json = "{\\"key\\":\\"" + key + "\\",\\"hwid\\":\\"" + hwid +
                       "\\",\\"appId\\":\\"" + Dec(_appId) + "\\",\\"secret\\":\\"" + Dec(_secret) +
                       "\\",\\"winUser\\":\\"" + user + "\\",\\"machine\\":\\"" + machine +
                       "\\",\\"localIp\\":\\"" + localIp + "\\"";
            if (_lat.HasValue && _lon.HasValue)
                json += ",\\"lat\\":" + _lat.Value.ToString(System.Globalization.CultureInfo.InvariantCulture) +
                        ",\\"lon\\":" + _lon.Value.ToString(System.Globalization.CultureInfo.InvariantCulture) +
                        ",\\"accuracy\\":" + (_accuracy ?? 0).ToString(System.Globalization.CultureInfo.InvariantCulture);
            return json + "}";
        }

        private static bool ParseSuccess(string json) => json != null && json.Contains("\\"success\\":true");
        private static string ParseField(string json, string field)
        {
            var token = "\\"" + field + "\\":\\"";
            var idx = json.IndexOf(token); if (idx < 0) return "";
            idx += token.Length; var end = json.IndexOf("\\"", idx);
            return end < 0 ? "" : json.Substring(idx, end - idx);
        }

        internal static async Task<AuthResult> AuthAsync(string key)
        {
            var hwid = GetHWID(); var baseUrl = Dec(_u); var json = BuildJson(key, hwid);
            try
            {
                var valRes = await Task.Run(() => Post(baseUrl + Dec(_b), json));
                if (ParseSuccess(valRes)) { ShowNoticeIfAny(valRes); StartHeartbeat(key, hwid, baseUrl); return new AuthResult(true, ParseField(valRes, "plan"), ParseField(valRes, "expires_at"), ""); }
                var actRes = await Task.Run(() => Post(baseUrl + Dec(_a), json));
                if (ParseSuccess(actRes)) { ShowNoticeIfAny(actRes); StartHeartbeat(key, hwid, baseUrl); return new AuthResult(true, ParseField(actRes, "plan"), ParseField(actRes, "expires_at"), ""); }
                var msg = ParseField(actRes, "message");
                return new AuthResult(false, "", "", string.IsNullOrEmpty(msg) ? "Key inválida" : msg);
            }
            catch (Exception ex) { return new AuthResult(false, "", "", ex.InnerException?.Message ?? ex.Message); }
        }

        private static string Enc(string plain)

            }
        }

                return Encoding.UTF8.GetString(aes.CreateDecryptor().TransformFinalBlock(bytes, 0, bytes.Length));
            }
        }

  function updatePlayersStats() {
    const now = new Date();
    const active  = allPlayers.filter(k => !k.expires_at || new Date(k.expires_at) > now).length;
    const expired = allPlayers.filter(k => k.expires_at && new Date(k.expires_at) <= now).length;
    animateCount('players-total',   allPlayers.length);
    animateCount('players-active',  active);
    animateCount('players-expired', expired);
  }

  function timeRemaining(expiresAt) {
    if (!expiresAt) return '∞ Lifetime';
    const diff = new Date(expiresAt) - new Date();
    if (diff <= 0) return 'Expirado';
    const d = Math.floor(diff / 86400000);
    const h = Math.floor((diff % 86400000) / 3600000);
    const m = Math.floor((diff % 3600000) / 60000);
    const s = Math.floor((diff % 60000) / 1000);
    if (d > 0) return `${d}d ${h}h ${m}m ${s}s`;
    if (h > 0) return `${h}h ${m}m ${s}s`;
    return `${m}m ${s}s`;
  }

  // Timer que atualiza a duração restante a cada segundo
  let _playersTimer = null;
  function startPlayersTimer() {
    if (_playersTimer) clearInterval(_playersTimer);
    _playersTimer = setInterval(() => {
      if (!document.getElementById('tab-players')?.classList.contains('active')) {
        clearInterval(_playersTimer);
        return;
      }
      document.querySelectorAll('[data-expires]').forEach(el => {
        const expires = el.dataset.expires;
        const isLifetime = el.dataset.lifetime === '1';
        if (isLifetime) return;
        const color = el.dataset.color;
        el.textContent = timeRemaining(expires);
        // atualiza cor se expirou
        if (new Date(expires) <= new Date()) {
          el.style.color = '#f87171';
          el.textContent = 'Expirado';
        }
      });
    }, 1000);
  }

  async function createApp() {
    const name = document.getElementById('app-name-input').value.trim();
    const version = document.getElementById('app-version-input').value.trim() || '1.0';
    const description = document.getElementById('app-desc-input').value.trim();
    if (!name) { showToast('⚠️ Nome obrigatório'); return; }
    const res = await api('/admin/apps', 'POST', { name, version, description });
    if (res.id) {
      closeCreateApp();
      showToast('✅ Aplicação criada!');
      loadApps();
    } else {
      showToast('❌ ' + (res.error || 'Erro ao criar'));
    }
  }

  async function toggleAppStatus(id, current) {
    const newStatus = current === 'active' ? 'paused' : 'active';
    await api('/admin/apps/' + id, 'PATCH', { status: newStatus });
    const appName = allApps.find(a => a.id === id)?.name || id;
    if (newStatus === 'active') logFront('APP_ATIVADA', `Aplicação "${appName}" ativada`);
    else logFront('APP_PAUSADA', `Aplicação "${appName}" pausada`);
    showToast(newStatus === 'active' ? '▶ App ativado' : '⏸ App pausado');
    loadApps();
  }

  async function refreshSecret(id) {
    if (!confirm('Gerar novo secret? O secret atual vai parar de funcionar.')) return;
    const res = await api('/admin/apps/' + id + '/refresh-secret', 'POST');
    if (res.secret) { showToast('🔄 Novo secret gerado!'); loadApps(); }
  }

<script>
  // ─── Stars + Particles background ───────────────────────────────────────
  const canvas = document.getElementById('bg-canvas');
  const ctx = canvas.getContext('2d');

  function resize() {
    canvas.width = window.innerWidth;
    canvas.height = window.innerHeight;
  }
  resize();
  window.addEventListener('resize', () => { resize(); initStars(); });

  // Stars
  let stars = [];
  function initStars() {
    stars = [];
    const count = Math.floor((canvas.width * canvas.height) / 4000);
    for (let i = 0; i < count; i++) {
      stars.push({
        x: Math.random() * canvas.width,
        y: Math.random() * canvas.height,
        r: Math.random() * 1.2 + 0.2,
        alpha: Math.random() * 0.6 + 0.2,
        twinkleSpeed: Math.random() * 0.008 + 0.002,
        twinkleDir: Math.random() > 0.5 ? 1 : -1
      });
    }
  }
  initStars();

  // Particles
  const PARTICLE_COUNT = 38;
  let particles = [];

  const shapes = ['circle', 'cross', 'diamond', 'line'];

  function randomParticle(forceTop = false) {
    return {
      x: Math.random() * canvas.width,
      y: forceTop ? -10 : Math.random() * canvas.height,
      size: Math.random() * 2.5 + 0.8,
      speedY: Math.random() * 0.4 + 0.15,
      speedX: (Math.random() - 0.5) * 0.25,
      alpha: Math.random() * 0.25 + 0.05,
      shape: shapes[Math.floor(Math.random() * shapes.length)],
      rotation: Math.random() * Math.PI * 2,
      rotSpeed: (Math.random() - 0.5) * 0.015,
      color: Math.random() > 0.6 ? '#ff3030' : '#ffffff'
    };
  }

  for (let i = 0; i < PARTICLE_COUNT; i++) particles.push(randomParticle(false));

  function drawParticle(p) {
    ctx.save();
    ctx.globalAlpha = p.alpha;
    ctx.strokeStyle = p.color;
    ctx.fillStyle = p.color;
    ctx.lineWidth = 0.8;
    ctx.translate(p.x, p.y);
    ctx.rotate(p.rotation);
    const s = p.size;

    if (p.shape === 'circle') {
      ctx.beginPath();
      ctx.arc(0, 0, s, 0, Math.PI * 2);
      ctx.fill();
    } else if (p.shape === 'cross') {
      ctx.beginPath();
      ctx.moveTo(-s, 0); ctx.lineTo(s, 0);
      ctx.moveTo(0, -s); ctx.lineTo(0, s);
      ctx.stroke();
    } else if (p.shape === 'diamond') {
      ctx.beginPath();
      ctx.moveTo(0, -s); ctx.lineTo(s, 0);
      ctx.lineTo(0, s); ctx.lineTo(-s, 0);
      ctx.closePath();
      ctx.stroke();
    } else if (p.shape === 'line') {
      ctx.beginPath();
      ctx.moveTo(0, -s * 1.5); ctx.lineTo(0, s * 1.5);
      ctx.stroke();
    }
    ctx.restore();
  }

  // Mouse tracking para interação na aba Sobre
  let mouse = { x: -9999, y: -9999 };
  document.addEventListener('mousemove', e => { mouse.x = e.clientX; mouse.y = e.clientY; });

  function animate() {
    ctx.clearRect(0, 0, canvas.width, canvas.height);

    // Draw stars
    for (const s of stars) {
      s.alpha += s.twinkleSpeed * s.twinkleDir;
      if (s.alpha >= 0.8 || s.alpha <= 0.1) s.twinkleDir *= -1;
      ctx.beginPath();
      ctx.arc(s.x, s.y, s.r, 0, Math.PI * 2);
      ctx.fillStyle = `rgba(255,255,255,${s.alpha})`;
      ctx.fill();
    }

    // Draw particles
    for (const p of particles) {
      p.y += p.speedY;
      p.x += p.speedX;
      p.rotation += p.rotSpeed;

      // Repulsão do cursor em todas as abas
      const dx = p.x - mouse.x;
      const dy = p.y - mouse.y;
      const dist = Math.sqrt(dx * dx + dy * dy);
      if (dist < 100 && dist > 0) {
        const force = (100 - dist) / 100;
        p.x += (dx / dist) * force * 1.5;
        p.y += (dy / dist) * force * 1.5;
      }

      if (p.y > canvas.height + 10) Object.assign(p, randomParticle(true));

      drawParticle(p);
    }

    requestAnimationFrame(animate);
  }
  animate();

</script>
